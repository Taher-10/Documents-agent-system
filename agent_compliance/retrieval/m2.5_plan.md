**Non-mappable classifier:**

```python
_NON_MAPPABLE_TITLE_PATTERNS = [
    "historique", "révisions", "modifications",
    "table des matières", "sommaire",
    "objet du document", "page de garde",
    "liste de diffusion", "approbation", "signatures",
    "amendment", "change history", "table of contents",
]
_SHORT_TEXT_THRESHOLD = 120

def _is_mappable(section) -> bool:
    if any(pat in section.title.lower() for pat in _NON_MAPPABLE_TITLE_PATTERNS):
        return False
    if len(section.raw_text.strip()) < _SHORT_TEXT_THRESHOLD:
        return False
    return True
```

**Prompt renderers — pure functions, no LLM:**

```python
def _render_clause_menu(menu: dict[str, list[tuple[str, str]]]) -> str:
    lines = []
    for norm_id, clauses in menu.items():
        lines.append(norm_id + ":")
        for number, title in clauses:
            lines.append(f"  {number}  {title}")
    return "\n".join(lines)

def _mapping_prompt(section, menu_text: str, doc_type: str, doc_level: int) -> list:
    from langchain_core.messages import SystemMessage, HumanMessage
    return [
        SystemMessage(content=(
            "You are a QHSE compliance analyst. Given a document section and a list of "
            "ISO clause numbers with titles, identify which clauses this section addresses. "
            "Output only clause numbers that genuinely relate to the section content. "
            "Output 3 to 6 clause numbers maximum."
        )),
        HumanMessage(content=(
            f"Document type: {doc_type}  Level: {doc_level}\n"
            f"Section type: {section.section_type.value} (hint — do not restrict to this type only)\n"
            f"Section title: {section.title}\n"
            f"Section text:\n{section.raw_text}\n\n"
            f"ISO clause menu:\n{menu_text}"
        )),
    ]

def _assessment_prompt(section, clauses: list) -> list:
    from langchain_core.messages import SystemMessage, HumanMessage
    clause_block = "\n---\n".join(
        f"[{c.norm_id}] {c.clause_number} — {c.clause_title}\n{c.text}"
        for c in clauses
    )
    return [
        SystemMessage(content=(
            "You are a QHSE compliance analyst. Assess whether the document section "
            "satisfies each ISO clause provided. For each clause:\n"
            "- Quote evidence directly from the section text (verbatim).\n"
            "- State specifically what is missing or non-conforming.\n"
            "- Give concrete advice: what must be added, clarified, or documented.\n\n"
            "Rules:\n"
            "- evidence_text must be a verbatim quote from the section text.\n"
            "- advice must be specific: not 'incomplete documentation' but "
            "  'Add documented criteria for re-evaluation of suppliers'.\n"
            "- Only reference the clause numbers provided to you."
        )),
        HumanMessage(content=(
            f"Section type: {section.section_type.value}\n"
            f"Section title: {section.title}\n"
            f"Section text:\n{section.raw_text}\n\n"
            f"ISO clauses to assess:\n{clause_block}"
        )),
    ]
```

**Main node — all ISO lookups use `db_path`, not qdrant:**

```python
import logging
from functools import partial
from agent_compliance.graph.state import ComplianceState
from agent_compliance.graph.models import (
    MappingOutput, SectionMatchOutput, SectionMatch, to_section_match
)
from agent_compliance.retrieval.clause_store import (
    fetch_clauses_by_ids, fetch_clauses_by_section_type
)
from agent_compliance.graph.llm import get_llm

logger = logging.getLogger(__name__)

def react_mapper_node(state: ComplianceState, db_path: str) -> dict:
    llm = get_llm()
    section_matches: list[SectionMatch] = []

    for section in state["sections"]:

        if not _is_mappable(section):
            section_matches.append(SectionMatch(
                section_id=section.id, section_type=section.section_type.value,
                title=section.title, matched_clauses=[], status="NOT_APPLICABLE",
                gaps=[], confidence=1.0, has_commitments=False,
            ))
            continue

        try:
            # Call 1 — mapping
            menu_text = _render_clause_menu(state["clause_menu"])
            mapping: MappingOutput = llm.with_structured_output(MappingOutput).invoke(
                _mapping_prompt(section, menu_text, ...)  # fill doc_type/doc_level from state
            )

            # Validate — drop hallucinated IDs (SQLite returns [] for unknown IDs)
            fetched = fetch_clauses_by_ids(
                mapping.clause_ids, state["applicable_norms"],
                language=state["language"], db_path=db_path,
            )

            # Fallback
            if not fetched:
                logger.warning("0 valid IDs for section %s — using fallback", section.id)
                fetched = fetch_clauses_by_section_type(
                    section.section_type.value, state["applicable_norms"],
                    language=state["language"], db_path=db_path,
                )

            # Call 2 — assessment
            output: SectionMatchOutput = llm.with_structured_output(SectionMatchOutput).invoke(
                _assessment_prompt(section, fetched)
            )

            # Low-confidence downgrade
            if not mapping.clause_ids and output.confidence < 0.3:
                output.status = "NOT_APPLICABLE"

            has_commitments = any(
                term in section.raw_text.lower() for term in MODAL_TERMS_FR
            )
            section_matches.append(to_section_match(output, fetched, section, has_commitments))

        except Exception:
            logger.exception("Parse/LLM error on section %s", section.id)
            section_matches.append(SectionMatch(
                section_id=section.id, section_type=section.section_type.value,
                title=section.title, matched_clauses=[], status="MISSING",
                gaps=["llm_parse_error"], confidence=0.0, has_commitments=False,
            ))

    return {"section_matches": section_matches}
```
